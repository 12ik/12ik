  5. PHP中文字符串翻转
    function cstrrev($str) { $len = strlen($str); for($i = 0; $i < $len; $i++) { $char = $str{0}; if(ord($char) > 127) { $i++; if($i < $len) { $arr[] = substr($str, 0, 2); $str = substr($str, 2); } } else { $arr[] = $char; $str = substr($str, 1); } } return join(array_reverse($arr)); }
